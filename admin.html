function verEmergencias() { let data = JSON.parse(localStorage.getItem("emergencias")) || []; let html = ""; data.forEach((e, i) => { html += `
${e.user} (${e.role})
📍 ${e.lat}, ${e.lon}
⏰ ${e.time}
Estado: ${e.estado}
`; }); document.getElementById("data").innerHTML = html; } function resolver(i) { let data = JSON.parse(localStorage.getItem("emergencias")) || []; data[i].estado = "resuelto"; localStorage.setItem("emergencias", JSON.stringify(data)); verEmergencias(); }
